Ejemplo de clase privada

Estudiar! es una orden y peticion.

        class MiClase {
            constructor() {
              this._propiedadPrivada = 42;
            }
          
            obtenerPropiedadPrivada() {
              return this._propiedadPrivada;
            }
          
            establecerPropiedadPrivada(valor) {
              this._propiedadPrivada = valor;
            }
          }
          
          const instancia = new MiClase();
          console.log(instancia.obtenerPropiedadPrivada()); // 42
          instancia.establecerPropiedadPrivada(50);
          console.log(instancia.obtenerPropiedadPrivada()); // 50
          console.log(instancia._propiedadPrivada); // Aún se puede acceder, pero se considera una convención privada
